1.2. Motion in a central potental



var("m")

V = function("V")


def L_polar(m, V):
    def Lagrangian(local):
        r, phi = coordinate(local).list()
        rdot, phidot = velocity(local).list()
        T = 1 / 2 * m * (square(rdot) + square(r * phidot))
        return T - V(r)

    return Lagrangian

q = column_path([literal_function("r"), literal_function("phi")])
